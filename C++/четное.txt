int main()
{
    int x, y, z;
    cin >> x >> y >> z; ///Ввод 3 чисел
    if (x!=y) /// проверка на равенство чисел
    {
        cout << " цифры этого числа неодинаковы " << x << y << z << endl;
    }
    else
    {
        if (y!=z)
        {
            cout << " цифры этого числа неодинаковы " << x << y << z << endl;
        }
        else 
        {
            cout << "цифры этого числа одинаковы " << x << y << z << endl;
        }
    }
    
    
    return 0;
}
